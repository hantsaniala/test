[docs]    def build_attr2(self):
        '''
        New version of attributes translator.

        **Note :** If there is a char style rule in the paragraph, we will just skip it.
        '''

        prio = PriorityRule(self._css)
        self._span_list = self._tree.xpath('//span')

        if self._span_list is not []:
            for span in self._span_list:
                # FIXME : Be aware, sometimes vertical align may be a pixel value not
                # only text.
                is_sup = is_sub = prio.priority(span, 'vertical-align')
                is_italic = prio.priority(span, 'font-style')
                is_bold = prio.priority(span, 'font-weight')
                is_underline = prio.priority(span, 'text-decoration')
                is_upper = is_low = prio.priority(span, 'text-transform')

                # Small-caps value is not constant.
                is_sc = prio.priority(span, 'font-size')
                is_sc2 = prio.priority(span, 'font-variant')

                # These condition order must be lie this because of the
                # priority of the tag order if there are many attributes in the
                # same tag. The priority order is sc | up | lo > u > b > i >
                # sup | sub

                if is_sc is not None:
                    # This value is not fixed, lets figure out the best way to change
                    # it dynamicaly depend of the file value.
                    #
                    # WARNING : Risk of having footnote with an small-caps hi
                    # if is_sc == '75%' or is_sc == '83%':
                    if is_sc == '75%':
                        if span.text is not None:
                            span.text = span.text.lower()
                        self._khtag.add_parent(span, 'hi', 'rend', 'sc')

                if is_sc2 is not None:
                    if is_sc2 == 'small-caps':
                        if span.text is not None:
                            span.text = span.text.lower()
                        self._khtag.add_parent(span, 'hi', 'rend', 'sc')

                if is_upper is not None:
                    if is_upper == 'uppercase':
                        self._khtag.add_parent(span, 'hi', 'rend', 'up')

                if is_low is not None:
                    if is_low == 'lowercase':
                        self._khtag.add_parent(span, 'hi', 'rend', 'lw')

                if is_underline is not None:
                    if is_underline == 'underline':
                        self._khtag.add_parent(span, 'hi', 'rend', 'u')

                if is_bold is not None:
                    if is_bold == 'bold':
                        self._khtag.add_parent(span, 'hi', 'rend', 'b')

                if is_italic is not None:
                    if is_italic == 'italic':
                        self._khtag.add_parent(span, 'hi', 'rend', 'i')

                if is_sup is not None:
                    if is_sup == 'top' or is_sup == 'super':
                        self._khtag.add_parent(span, 'hi', 'rend', 'sup')

                if is_sub is not None:
                    if is_sub == 'bottom':
                        self._khtag.add_parent(span, 'hi', 'rend', 'sub')

                            a.tag = 'apnb'

                                r'..\/Text\/([\w\d\-\.]+)#(?P<id>[\w\d-]+)', a.attrib['href'])
                            if i is not None:
                                a.attrib['id'] = i.group('id')
                                # a.text = ''
                                a.attrib.pop('href', None)
                                a.attrib.pop('class', None)
                        else:
                            pass
                    else:
                        a.tag = 'ref'
                        a.attrib['target'] = a.attrib['href']
                        a.attrib.pop('href', None)

                elif 'id' in a.attrib:
                    if re.match(r'_idIndexMarker', a.attrib['id']) != None:
                        a.tag = 'renvlnk'
            except Exception as e:
                logging.warn(e)

                    fig.attrib.pop(att, None)

    '''
    Implemeted tool for working with tag and simplifing the using of the tag.
    '''

    def __init__(self, tree):
        self._tree = tree
        self._assossiated_list = {}

    def _get_astd(self, tag, tag_dict):
        '''
        Get the assossiation between the xml-tag and the tag itself. So if we had these relationship, we can iterate throw all tag to update these assossiated tag with the respected value.
        '''
        tag_dict = tag_dict
        rgx = re.compile(r"xml-([\w-]+)")
        if 'class' in tag.attrib:
            logging.info(tag.attrib['class'])
            try:
                logging.info('Checking if had many value')
                if tag.attrib['class'].split(' ')[1]:
                    logging.info('Ok it has')
                    for t in tag.attrib['class'].split(' '):
                        logging.info(t)
                        logging.info('Lauching the matching process')
                        srch_res = re.match(rgx, t)
                        logging.debug('srch_res : ' + str(srch_res))
                        if srch_res is not None:

                            logging.info('Match !')

                            # Append key and value if exist or create if not.
                            # {
                            #   xml-tag:
                            #   {
                            #     tag: [class]
                            #   }
                            # }
                            #
                            # Source :
                            # http://stackoverflow.com/a/12906014/5527968

                            tag_class = (tag.attrib['class'].split(' ')[0])

                            if tag_dict[srch_res.group(1)].get(tag.tag):
                                tag_dict[srch_res.group(1)][
                                    tag.tag].append(tag_class)
                            else:
                                tag_dict[srch_res.group(1)][tag.tag] = [
                                    tag_class]

                            self._assossiated_list = tag_dict
                        else:
                            continue
                else:
                    # FIXME : If only one value from the class.
                    logging.error('Tag cannot be only one.')

            except Exception as e:
                logging.info(e)
                pass

[docs]    def update_tag(self, ass_list):
        '''
        Get the list of assossiation tag and transform one by one all tag to theire respective assossiation with attributes.
        '''
        for xml_tag in ass_list:

            # Heading level
            if xml_tag == 'part':
                self._update_xhtml_tag(ass_list, xml_tag, 'head')
            if xml_tag == 'chapter':
                self._update_xhtml_tag(ass_list, xml_tag, 'head', {
                    'rendition': '#chaptertitle'})
            if xml_tag == 'stitre1':
                self._update_xhtml_tag(ass_list, xml_tag, 'head', {
                    'rendition': '#sectiontitle'})
            if xml_tag == 'stitre2':
                self._update_xhtml_tag(ass_list, xml_tag, 'head', {
                    'rendition': '#ssectiontitle'})
            if xml_tag == 'stitre3':
                self._update_xhtml_tag(ass_list, xml_tag, 'head', {
                    'rendition': '#sssectiontitle'})
            if xml_tag == 'stitre4':
                self._update_xhtml_tag(ass_list, xml_tag, 'head', {
                    'rendition': '#ssssectiontitle'})
            # Only stop at this level for now.
            # We will see later if more level
            # is needed.

            # Table
            # Note
            # Image

            # Text bloc
            if xml_tag == 'caption':
                self._update_xhtml_tag(ass_list, xml_tag, 'caption')
            if xml_tag == 'author':
                self._update_xhtml_tag(ass_list, xml_tag, 'author')


    def _update_xhtml_tag(self, ass_list, xml_tag, new_tag_name, att={}):
        '''
        Get the xhtml tag and rename the tag to their respective name and attribute.

        :param dict att: Liste of attribute.
        '''
        for xhtml_tag in ass_list[xml_tag]:
            element_lst = []
            for xhtml_css in set(ass_list[xml_tag][xhtml_tag]):
                xpath_val = '//' + xhtml_tag + \
                    '[contains(@class, "' + xhtml_css + '")]'
                for element in self._tree.xpath(str(xpath_val)):
                    element.tag = new_tag_name

                    # If some attribute are specified
                    if att != {}:
                        for val in att:
                            element.attrib[val] = att[val]

                    # Remove the class and style attribute after the
                    # conversion
                    element.attrib.pop('class', None)
                    element.attrib.pop('style', None)
