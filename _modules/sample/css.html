    def __init__(self, path):
        logging.info('Parsing css file')
        self._path = path

        self._css = []
        self._styles = ""

        self._css_list = self.generate_list()
        logging.debug(self._css_list)
        self._parser = tinycss.make_parser()

        for css_file in self._css_list:
            with open(css_file) as f:
                self._styles += f.read()

        self._parsed = self._parser.parse_stylesheet(self._styles)
        self.parse(self._parsed)

[docs]    def unit_parse(self, value):
        '''
        Get the unit.

        :return: Parsed unit
        :rtype: str or None
        '''
        rgx = re.compile(
            r"(?P<ent>([0-9])+)(\.(?P<dec>([0-9]+))?)(?P<unit>(px|em|pt|\%))")
        res = re.match(rgx, value)

        if res is not None:
            return {res.group('ent'), res.group('dec'), res.group('unit')}



[docs]    def parse(self, parsed):
        '''
        Get the file, parse and return the CSS object as list.

        :param obj parsed: Parsed rule from CSS file as Object
        '''
        if parsed.rules is not None:
            for rule in parsed.rules:
                # logging.debug(''.join(t.as_css() for t in rule.selector))
                if rule.selector is not None:
                    for t in rule.selector:
                        self._r = CSSRule()
                        # logging.debug(t.as_css())
                        self._r.name = t.as_css()
                        for dec in rule.declarations:
                            # logging.debug('\t{0}: {1}'.format(dec.name,
                            # dec.value.as_css()))
                            self._r.attrib = (dec.name, dec.value.as_css())
                        self._css.append(self._r)

[docs]    def priority(self, tag, rule_dec='text-align'):
        '''
        Get the priority from style or class if they exist.

        :return: alignement
        :rtype: str
        '''

        # By default, the priority is for the style attribute found in the tag.
        # So if the style attribute exist, we must first check if the needed
        # value can be found here. If not, we will check the class attribute if
        # exist and find if there is the needed param in it too. After we will
        # check the priority of the class rule if there are many because the
        # last rule is always the priority. But if we cannot find it in the
        # last, we must check the previous rule if there is in.
        #
        # STRUCT :
        #
        # if has style
        #   if need_param in style
        #       get need_value from style
        #   elif has class
        #       if need_param in class
        #           get need_value from class
        #       else
        #           set default_value
        #   else
        #       set default_value
        # elif has class
        #   if need_param in class
        #       get need_value from class
        #   else
        #       set default_value
        # else
        #   set default_value

        if self.has_style(tag):
            logging.debug(str(tag) + ' has style...')
            ePub_stl = ePubStyle(tag)
            logging.debug('Has a declaration : ' + str(ePub_stl.has_dec()))
            
            # FIXME : Value still False. Semantic error.
            return ePub_stl.declaration(rule_dec)

        elif self.has_class(tag):
            logging.debug(str(tag) + ' has class...')
            ePub_cls = ePubClass(tag, self._css)
            return ePub_cls.declaration(rule_dec)

        else:
            logging.debug('Daaaaahhhh.... No needed declaration in it too...')
            return self.set_to_default(tag)

        '''
        Get the priority of the class declaration and return the result.
        
        :return: declaration
        :rtype: str
        '''
        if self.has_many_rule():
            logging.debug('Grr... it has many rule (T T)')
            self._r_list = self.get_rule_list()
            logging.debug(self._r_list)
            for self._r in self._r_list[::-1]:
                logging.debug(self._r)
                self.find_dec(self._r, rule_dec)
                if self._has_dec:
                    logging.debug('Found the declaration from many rule :D')
                    return self.get_dec()
                else:
                    logging.debug('Not declaration found on it... ')
                    continue
        elif self.has_many_rule() is False:
            if 'class' in self._tag.attrib:
                logging.debug('So it has only one class rule :D')
                self.find_dec(self._tag.attrib['class'], rule_dec)
                if self.has_dec():
                    logging.debug('Found the declaration :D')
                    return self.get_dec()
                else:
                    logging.debug('Setting it to default value..')
                    return self.set_to_default()
            else:
                logging.debug('Setting it to default value..')
                return self.set_to_default()
        else:
            logging.debug('Setting it to default value..')
            return self.set_to_default()

        logging.debug('rule_dec : '+rule_dec)
        self._rule_dec = rule_dec
        try:
            if 'style' in self._tag.attrib:
                self._stl_dec = self._tag.attrib['style']
                logging.debug('Style declaration : ' + str(self._stl_dec))
                self._style_match_rgx = re.compile(
                    r'(?P<dec>[\w-]+):(?P<val>([a-z\-]+)|(?P<num>[0-9\.]+)(?P<unit>cm|em|p[xt]|%?))(;?)')
                self._style_match_res = re.findall(

                logging.debug('Match result : ' + str(self._style_match_res))

                            logging.debug(x)
                            self._has_dec = True
                            self._dec = x[1]
                            logging.debug('Declaration : ' + str(self._dec))
                            return self._dec
                        else:
                            continue

                pass

    '''

    :return: CSSRule
    '''

    def __init__(self):
        self._name = ""
        self._attrib = {}

    @property
    def name(self):
        '''
        The Rule name as string.

        :return: CSSRule.name
        '''
        return self._name

    @name.setter
    def name(self, name):
        self._name = name

    @property
    def attrib(self):
        '''
        The Rule attribute as string.

        :return: CSSRule.attrib
        '''
        return self._attrib

    @attrib.setter
    def attrib(self, att):
        self._attrib[att[0]] = att[1]
