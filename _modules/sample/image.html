[docs]    def copy_to(self, dest):
        self._dest_dir = dest
        self._generate_list()

        logging.info("Copying pictures")
        if self._jpg_list != []:
            for jpg in self._jpg_list:
                logging.info(
                    "Copying {} to {}".format(os.path.split(jpg)[1], self._dest_dir))
                if not os.path.exists(self._dest_dir):
                    os.mkdir(self._dest_dir)
                shutil.copy(jpg, os.path.join(self._dest_dir))

        if self._jpg_list != []:
            for png in self._png_list:
                logging.info(
                    "Copying {} to {}".format(os.path.split(png)[1], self._dest_dir))
                if not os.path.exists(self._dest_dir):
                    os.mkdir(self._dest_dir)
                shutil.copy(png, os.path.join(self._dest_dir))


    def _generate_list(self):
        logging.info("Generate pictures list")

        self._jpg_list = glob.glob(os.path.join(
            self._path.tmp_dir, "OEBPS", "Images", "*.jpg"))
        self._png_list = glob.glob(os.path.join(
            self._path.tmp_dir, "OEBPS", "Images", "*.png"))
