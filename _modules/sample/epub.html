[docs]    def extract(self):
        if os.path.exists(self._path.tmp_dir):
            logging.info("Deleting the folder {}".format(self._path.tmp_dir))
            shutil.rmtree(self._path.tmp_dir)

        with zipfile.ZipFile(os.path.join(self._path.fpath, self._path.fname_w_ext), 'r') as z:
            z.extractall(self._path.tmp_dir)
